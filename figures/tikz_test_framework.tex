% Requires: \usepackage{tikz}
% TikZ libraries: \usetikzlibrary{arrows.meta,positioning,calc}

\colorlet{frameworkFill}{gray!10}
\colorlet{lineColor}{black}

\definecolor{mplBlue}{HTML}{1F77B4}
\definecolor{mplGreen}{HTML}{2CA02C}
\definecolor{mplOrange}{HTML}{FF7F0E}

\colorlet{suiteFill}{mplBlue!35}
\colorlet{caseFill}{mplOrange!35}
\colorlet{trialFill}{mplGreen!35}

% \colorlet{suiteFill}{blue!50}
% \colorlet{caseFill}{green!50}
% \colorlet{trialFill}{orange!50}

\newcommand{\dotsYoffset}{2mm}
\newcommand{\levelYspacing}{8mm}
\newcommand{\busYspacing}{4mm}
\newcommand{\suiteXspacing}{4mm}
\newcommand{\levelXspacing}{16mm}

\begin{tikzpicture}[
  font=\small,
  >=Latex,
  box/.style={
    draw,
    rounded corners=2pt,
    align=center,
    inner xsep=6pt,
    inner ysep=5pt
  },
  root/.style={
    box,
    thick,
    fill=frameworkFill,
    minimum width=56mm,
    minimum height=10mm
  },
  suite/.style={
    box,
    fill=suiteFill,
    thick,
    minimum width=12mm,
    minimum height=10mm
  },
  case/.style={
    box,
    fill=caseFill,
    thick,
    minimum width=24mm,
    minimum height=9mm
  },
  trial/.style={
    box,
    fill=trialFill,
    thick,
    minimum width=24mm,
    minimum height=9mm
  },
  dots/.style={font=\large, text=lineColor},
  line/.style={thick, draw=lineColor},
  arrow/.style={-Latex, thick, draw=lineColor}
]

% Framework level
\node[root] (framework) at (0,0) {Test Framework};

% Suite level
\coordinate (suiteLevel) at ($(framework.south)+(0,-\levelYspacing)$);
\node[suite, anchor=north] (suite3) at (framework |- suiteLevel)
  {$n=3$};
\node[suite, anchor=north east] (suite2) at ($(suite3.north west)+(-\suiteXspacing,0)$)
  {Test Suite for\\[-1pt] 
  $n=2$ leaks\\[-1pt]
  (\autoref{alg:test_suite})};
\node[suite, anchor=north west] (suite4) at ($(suite3.north east)+(\suiteXspacing,0)$)
  {$n=4$};
\node[suite, anchor=north west] (suite5) at ($(suite4.north east)+(\suiteXspacing,0)$)
  {$n=5$};

\coordinate (suiteBus) at ($(framework.south)+(0,-\busYspacing)$);
\draw[line] (framework.south) -- (suiteBus);
\draw[line] ($(suite2.north |- suiteBus)$) -- ($(suite5.north |- suiteBus)$);
\foreach \s in {suite2,suite3,suite4,suite5}
  \draw[arrow] ($(\s.north |- suiteBus)$) -- (\s.north);

% Case level for the n=2 suite
\coordinate (caseLevel) at ($(suite2.south)+(0,-\levelYspacing)$);
\node[case, anchor=north] (caseC) at (framework |- caseLevel)
  {Test Case: $c$\\[3pt]
   \begin{tabular}{@{}l@{}}
      - Sample $\mathbf{x}_c^\ast$ and $\theta$\\
      - Simulate $\mathcal{M}$
   \end{tabular}};
\node[case, anchor=north east] (case1) at ($(caseC.north west)+(-\levelXspacing,0)$)
  {Test Case: $1$};
\node[dots, anchor=north] (caseDotsA) at ($(case1.north east)!0.5!(caseC.north west)+(0,-\dotsYoffset)$) {$\cdots$};
\node[case, anchor=north west] (caseN) at ($(caseC.north east)+(\levelXspacing,0)$)
  {Test Case: $N_{\mathrm{cases}}$};
\node[dots, anchor=north] (caseDotsB) at ($(caseC.north east)!0.5!(caseN.north west)+(0,-\dotsYoffset)$) {$\cdots$};

\coordinate (caseBus) at ($(suite2.south)+(0,-\busYspacing)$);
\draw[line] (suite2.south) -- (caseBus);
\draw[line] ($(case1.north |- caseBus)$) -- ($(caseN.north |- caseBus)$);
\foreach \c in {case1,caseC,caseN}
  \draw[arrow] ($(\c.north |- caseBus)$) -- (\c.north);

% Trial level inside the representative case
\node[trial, below=\levelYspacing of caseC] (trialT)
  {Trial: $t$\\[3pt]
   \begin{tabular}{@{}l@{}}
      - Sample $\mathbf{x}_0$\\
      - $\mathrm{Solve}(\mathcal{M},\theta,\mathbf{x}_0)$
   \end{tabular}};
\node[trial, anchor=north] (trial1) at (case1 |- trialT.north)
  {Trial: $1$};
\node[dots, anchor=north] (trialDotsA) at ($(caseDotsA |- trialT.north)+(0,-\dotsYoffset)$) {$\cdots$};
\node[trial, anchor=north] (trialN) at (caseN |- trialT.north)
  {Trial: $N_{\mathrm{trials}}$};
\node[dots, anchor=north] (trialDotsB) at ($(caseDotsB |- trialT.north)+(0,-\dotsYoffset)$) {$\cdots$};

\coordinate (trialBus) at ($(caseC.south)+(0,-\busYspacing)$);
\draw[line] (caseC.south) -- (trialBus);
\draw[line] ($(trial1.north |- trialBus)$) -- ($(trialN.north |- trialBus)$);
\foreach \t in {trial1,trialT,trialN}
  \draw[arrow] ($(\t.north |- trialBus)$) -- (\t.north);
\end{tikzpicture}
